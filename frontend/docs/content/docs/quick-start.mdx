# Quick Start

This guide will walk you through creating your first web application with Forge Go. In just a few minutes, you'll have a working web server with routing, middleware, and basic functionality.

<Callout type="info">
This guide assumes you have already [installed Forge](/docs/go/getting-started/installation) and have Go 1.21+ installed.
</Callout>

## Project Setup

Let's create a new Go project and set up Forge:

```bash
# Create a new directory for your project
mkdir my-forge-app && cd my-forge-app

# Initialize Go module
go mod init my-forge-app

# Install Forge
go get github.com/xraph/forge

# Install database driver (optional, choose one)
go get github.com/lib/pq              # PostgreSQL
go get github.com/go-sql-driver/mysql # MySQL
go get modernc.org/sqlite             # SQLite
```

## Basic Configuration

Create a `main.go` file with the basic Forge setup:

```go
package main

import (
    "log"
    "os"

    "github.com/xraph/forge"
)

func main() {
    // Initialize Forge app
    app := forge.New(forge.Config{
        Name:    "My Forge App",
        Version: "1.0.0",
    })

    // Add middleware
    app.Use(forge.Logger())
    app.Use(forge.Recovery())

    // Add a simple home route
    app.GET("/", func(c *forge.Context) error {
        return c.JSON(200, map[string]string{
            "message": "Welcome to Forge!",
            "version": "1.0.0",
        })
    })

    // Add health check route
    app.GET("/health", func(c *forge.Context) error {
        return c.JSON(200, map[string]string{
            "status": "healthy",
        })
    })

    // Add API routes
    api := app.Group("/api")
    api.GET("/status", func(c *forge.Context) error {
        return c.JSON(200, map[string]interface{}{
            "status":    "running",
            "timestamp": forge.Now(),
        })
    })

    // Start the server
    port := getEnv("PORT", "8080")
    log.Printf("ðŸš€ Server starting on http://localhost:%s", port)
    log.Fatal(app.Listen(":" + port))
}

// Helper function to get environment variables with defaults
func getEnv(key, defaultValue string) string {
    if value := os.Getenv(key); value != "" {
        return value
    }
    return defaultValue
}
```

## Environment Configuration

Create a `.env` file for your environment variables:

```bash
# Server configuration
PORT=8080
HOST=localhost

# Database configuration (optional)
DATABASE_URL=postgres://user:pass@localhost/forge_db?sslmode=disable

# Optional: Redis for caching
REDIS_URL=redis://localhost:6379

# Optional: Logging configuration
LOG_LEVEL=info
LOG_FORMAT=json
```

<Callout type="warn">
**Security Note**: Never commit your `.env` file to version control. Add it to your `.gitignore` file.
</Callout>

Run your Forge application:

## Adding Middleware

Enhance your application with Forge middleware:

```go
package main

import (
    "log"
    "os"
    "time"

    "github.com/xraph/forge"
)

func main() {
    app := forge.New(forge.Config{
        Name:    "My Forge App",
        Version: "1.0.0",
    })

    // Global middleware
    app.Use(forge.Logger())
    app.Use(forge.Recovery())
    app.Use(forge.CORS())

    // Rate limiting middleware
    app.Use(forge.RateLimit(forge.RateLimitConfig{
        Max:     100,
        Window:  time.Minute,
        Message: "Too many requests",
    }))

    // Public routes
    app.GET("/", func(c *forge.Context) error {
        return c.JSON(200, map[string]string{
            "message": "Welcome to Forge!",
            "version": "1.0.0",
        })
    })

    // API routes with additional middleware
    api := app.Group("/api")
    api.Use(forge.RequestID()) // Add request ID to each request

    api.GET("/status", func(c *forge.Context) error {
        return c.JSON(200, map[string]interface{}{
            "status":    "running",
            "timestamp": forge.Now(),
            "requestId": c.Get("requestId"),
        })
    })

    // Protected routes group
    protected := api.Group("/protected")
    protected.Use(forge.AuthRequired()) // Add authentication middleware

    protected.GET("/profile", func(c *forge.Context) error {
        return c.JSON(200, map[string]interface{}{
            "message": "This is a protected route",
            "user":    c.Get("user"),
        })
    })

    log.Println("ðŸš€ Server starting on http://localhost:8080")
    log.Fatal(app.Listen(":8080"))
}
```

## Testing Middleware

Test the middleware functionality:

```bash
# Test rate limiting (make multiple requests quickly)
for i in {1..5}; do curl -X GET http://localhost:8080/; done

# Test request ID middleware
curl -X GET http://localhost:8080/api/status

# Test protected route (should return 401 Unauthorized)
curl -X GET http://localhost:8080/api/protected/profile

# Test CORS (from browser console or with -H Origin header)
curl -X GET http://localhost:8080/ -H "Origin: http://localhost:3000"
```

## Adding Plugins

Let's add some plugins to enhance functionality:

```go
package main

import (
    "log"
    "os"

    "github.com/xraph/forge"
    "github.com/xraph/forge/plugins/database"
    "github.com/xraph/forge/plugins/observability"
)

func main() {
    app := forge.New(forge.Config{
        Name:    "My Forge App",
        Version: "1.0.0",
    })

    // Add database plugin
    dbPlugin := database.New(database.Config{
        Driver: "postgres",
        URL:    os.Getenv("DATABASE_URL"),
    })
    app.Use(dbPlugin)

    // Add observability plugin
    obsPlugin := observability.New(observability.Config{
        Metrics: true,
        Tracing: true,
        Logging: true,
    })
    app.Use(obsPlugin)

    // Your routes here...
    app.GET("/", func(c *forge.Context) error {
        return c.JSON(200, map[string]string{
            "message": "Welcome to Forge with plugins!",
        })
    })

    log.Println("ðŸš€ Server with plugins starting on http://localhost:8080")
    log.Fatal(app.Listen(":8080"))
}
```

## Frontend Integration

Here's a simple HTML page to test your Forge application:

```html
<!DOCTYPE html>
<html>
<head>
    <title>Forge Demo</title>
    <style>
        body { font-family: Arial, sans-serif; max-width: 800px; margin: 50px auto; padding: 20px; }
        .container { background: #f8f9fa; padding: 20px; border-radius: 8px; margin: 20px 0; }
        button { background: #007bff; color: white; padding: 10px 20px; border: none; cursor: pointer; margin: 5px; border-radius: 4px; }
        button:hover { background: #0056b3; }
        .response { background: #e9ecef; padding: 15px; border-radius: 4px; margin: 10px 0; font-family: monospace; }
        .error { color: #dc3545; }
        .success { color: #28a745; }
    </style>
</head>
<body>
    <h1>Forge Demo</h1>
    
    <div class="container">
        <h2>API Testing</h2>
        <button onclick="testHome()">Test Home Route</button>
        <button onclick="testHealth()">Test Health Check</button>
        <button onclick="testStatus()">Test API Status</button>
        <button onclick="testRateLimit()">Test Rate Limit</button>
        
        <div id="response" class="response" style="display: none;"></div>
    </div>

    <div class="container">
        <h2>Server Information</h2>
        <p>This page demonstrates how to interact with your Forge application.</p>
        <p>Make sure your Forge server is running on <code>http://localhost:8080</code></p>
    </div>

    <script>
        const API_BASE = 'http://localhost:8080';
        
        async function testHome() {
            try {
                const response = await fetch(`${API_BASE}/`);
                const data = await response.json();
                showResponse('Home Route', response.status, data);
            } catch (error) {
                showResponse('Home Route', 'Error', error.message, 'error');
            }
        }

        async function testHealth() {
            try {
                const response = await fetch(`${API_BASE}/health`);
                const data = await response.json();
                showResponse('Health Check', response.status, data);
            } catch (error) {
                showResponse('Health Check', 'Error', error.message, 'error');
            }
        }

        async function testStatus() {
            try {
                const response = await fetch(`${API_BASE}/api/status`);
                const data = await response.json();
                showResponse('API Status', response.status, data);
            } catch (error) {
                showResponse('API Status', 'Error', error.message, 'error');
            }
        }

        async function testRateLimit() {
            try {
                const promises = [];
                for (let i = 0; i < 5; i++) {
                    promises.push(fetch(`${API_BASE}/`));
                }
                const responses = await Promise.all(promises);
                const results = responses.map(r => ({ status: r.status, statusText: r.statusText }));
                showResponse('Rate Limit Test', 'Multiple Requests', results);
            } catch (error) {
                showResponse('Rate Limit Test', 'Error', error.message, 'error');
            }
        }

        function showResponse(endpoint, status, data, type = 'success') {
            const responseEl = document.getElementById('response');
            responseEl.style.display = 'block';
            responseEl.className = `response ${type}`;
            responseEl.innerHTML = `
                <strong>${endpoint}</strong><br>
                <strong>Status:</strong> ${status}<br>
                <strong>Response:</strong><br>
                <pre>${JSON.stringify(data, null, 2)}</pre>
            `;
        }
    </script>
</body>
</html>
```

Save this as `index.html` and serve it with a simple HTTP server:

```bash
# Python 3
python -m http.server 3000

# Node.js (if you have npx)
npx serve .

# Go (if you have a simple server)
go run -c "package main; import \"net/http\"; func main() { http.ListenAndServe(\":3000\", http.FileServer(http.Dir(\".\"))) }"
```

Visit `http://localhost:3000` to test your Forge application!
